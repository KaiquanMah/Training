def fizzBuzz(n):
    # Write your code here
    for i in range(1,n+1):
        if i%3 == 0:
            if i%5 != 0:
                print("Fizz")
            else:
                print("FizzBuzz")
        elif i%5 == 0:
            print("Buzz")
        else:
            print(i)
