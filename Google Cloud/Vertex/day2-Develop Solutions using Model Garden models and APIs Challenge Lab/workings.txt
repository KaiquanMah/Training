cd ~  
gcloud storage cp gs://qwiklabs-asl-01-752ad6b30635-bucket/model-garden-lab-app.zip .
unzip model-garden-lab-app.zip
cd model-garden-lab-app
python3 -m venv .venv
source .venv/bin/activate
pip install -r requirements.txt
export GOOGLE_CLOUD_PROJECT=qwiklabs-asl-01-752ad6b30635
export GOOGLE_CLOUD_REGION="us-central1"
export GOOGLE_CLOUD_CLAUDE_REGION="us-east5"
gcloud compute firewall-rules create allow-port-8088 \
    --allow tcp:8088 \
    --source-ranges=0.0.0.0/0 \
    --description="Allow TCP 8088 to this machine from any source"
